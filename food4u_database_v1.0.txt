create database food4u;
use food4u;
CREATE user 'fooduser'@'localhost' IDENTIFIED BY 'foodpass';
GRANT ALL PRIVILEGES ON * . * TO 'fooduser'@'localhost';
FLUSH PRIVILEGES;
create table user (userid INT NOT NULL AUTO_INCREMENT PRIMARY KEY, username VARCHAR(20) NOT NULL, 
password VARCHAR(255) NOT NULL, fname VARCHAR(20) NOT NULL, lname VARCHAR(30) NOT NULL, address VARCHAR(30) NOT NULL, img VARCHAR(255));
create table owner (ownerid INT NOT NULL AUTO_INCREMENT PRIMARY KEY, ownername VARCHAR(20) NOT NULL, 
password VARCHAR(255) NOT NULL, restaurant VARCHAR(30) NOT NULL, address VARCHAR(30) NOT NULL, img VARCHAR(255));
create table restaurant (restaurantid INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(255) NOT NULL, type VARCHAR(30) NOT NULL, pricerange VARCHAR(5) NOT NULL, ownerid int NOT NULL);
create table menu (menuid INT NOT NULL AUTO_INCREMENT PRIMARY KEY, dish int(5) NOT NULL, name VARCHAR(30) NOT NULL,
description VARCHAR(255) NOT NULL, price float  NOT NULL, img VARCHAR(255), restaurantid int NOT NULL);
